**Create src/components/editor/color-swatch-picker.tsx:**

```typescript
"use client";
import { motion } from "framer-motion";
import { THEME_PRESETS, PALETTE_DISPLAY_NAMES } from "@/lib/theme-presets";
import type { PaletteKey } from "@/types/site-theme";

interface ColorSwatchPickerProps {
  selected: PaletteKey;
  onChange: (key: PaletteKey) => void;
}

export function ColorSwatchPicker({ selected, onChange }: ColorSwatchPickerProps) {
  const keys = Object.keys(THEME_PRESETS) as PaletteKey[];
  return (
    <div className="grid grid-cols-4 gap-1">
      {keys.map((key) => (
        <button
          key={key}
          onClick={() => onChange(key)}
          aria-label={PALETTE_DISPLAY_NAMES[key]}
          className="w-11 h-11 flex items-center justify-center"
        >
          <motion.div
            whileHover={{ scale: 1.05 }}
            whileTap={{ scale: 0.95 }}
            animate={selected === key ? { scale: 1.1 } : { scale: 1.0 }}
            transition={{ duration: 0.15, ease: "easeOut" }}
            style={{
              width: 28,
              height: 28,
              borderRadius: "50%",
              backgroundColor: THEME_PRESETS[key],
              outline: selected === key ? `2px solid ${THEME_PRESETS[key]}` : "none",
              outlineOffset: "2px",
            }}
          />
        </button>
      ))}
    </div>
  );
}
```

```typescript
"use client";
import { motion } from "framer-motion";
import { FONT_FAMILY_MAP, FONT_DISPLAY_NAMES } from "@/lib/theme-presets";
import type { FontKey } from "@/types/site-theme";

interface FontPickerProps {
  selected: FontKey;
  onChange: (key: FontKey) => void;
  accentColor: string;  // hex string from THEME_PRESETS[activePalette] for underline
}

export function FontPicker({ selected, onChange, accentColor }: FontPickerProps) {
  const keys = Object.keys(FONT_FAMILY_MAP) as FontKey[];
  return (
    <div className="flex flex-col gap-0.5">
      {keys.map((key) => (
        <motion.button
          key={key}
          onClick={() => onChange(key)}
          whileHover={{ x: 2 }}
          transition={{ type: "spring", stiffness: 400, damping: 30 }}
          className="w-full text-left py-2 px-1 text-sm rounded transition-colors hover:bg-zinc-100"
          style={{
            fontFamily: FONT_FAMILY_MAP[key],
            fontWeight: selected === key ? 700 : 400,
            borderBottom: selected === key ? `2px solid ${accentColor}` : "2px solid transparent",
            color: selected === key ? "#18181b" : "#71717a",
          }}
        >
          {FONT_DISPLAY_NAMES[key]}
        </motion.button>
      ))}
    </div>
  );
}
```

**Part A — Create src/components/editor/appearance-tab.tsx:**

AppearanceTab receives `themeSettings`, `setThemeSettings`, and `previewRootRef` props. On swatch/font click, it immediately calls `applyThemePreview` to mutate CSS vars on the preview wrapper. It also calls `onSave` to trigger the editor's save function.

Exact prop interface:
```typescript
interface AppearanceTabProps {
  themeSettings: SiteTheme;
  setThemeSettings: (theme: SiteTheme) => void;
  previewRootRef: React.RefObject<HTMLDivElement | null>;
  onSave: () => void;
  isSaving: boolean;
}
```

The component:
1. Has a "Color" section heading (settings-label class, uppercase, 11px) with the ColorSwatchPicker
2. Has a "Font" section heading with the FontPicker
3. Has a "Save Appearance" button at the bottom using `<Button>` from `src/components/ui/button.tsx`
4. On palette change: calls `setThemeSettings({ ...themeSettings, palette: key })` AND immediately mutates the preview DOM via `previewRootRef.current?.style.setProperty("--site-accent", THEME_PRESETS[key])`; also sets `data-theme` attribute
5. On font change: calls `setThemeSettings({ ...themeSettings, font: key })` AND immediately mutates via `previewRootRef.current?.style.setProperty("--site-font-display", FONT_FAMILY_MAP[key])`

Section heading markup (matches existing `.settings-label` class from globals.css):
```tsx
<label className="settings-label">Color</label>

```
